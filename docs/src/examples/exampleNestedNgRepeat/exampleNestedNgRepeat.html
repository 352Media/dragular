Nested ngRepeat

Row {{$index}}
{{item.content}}
        
  // HTML

  <div ng-controller="Example15">
    <div ng-repeat="item in items" class='row'>
      <div class="row-handle">Row {{$index}}</div>
      <div ng-repeat="item in item.items" class="cell">{{item.content}}</div>
    </div>
  </div>
        
      
        
  // CSS

  .row {
    display: flex;
    flex-direction: row;
  }

  .cell {
    flex-grow: 1;
  }

  .row,
  .cell {
    margin: 10px;
    padding: 10px;
    background-color: rgba(0, 0, 0, 0.2);
  }
        
      
        
  // JS

  $timeout(function() { // timeount due to ngRepeat to be ready
    dragularService($element, {
      moves: function(el, container, handle) {
        return handle.classList.contains('row-handle');
      }
    });
    dragularService($element.children(), {
      moves: function(el, container, handle) {
        return !handle.classList.contains('row-handle');
      }
    });
  }, 0);
  $scope.items = [{
    items: [{
      content: 'Item a1'
    }, {
      content: 'Item a2'
    }, {
      content: 'Item a3'
    }, {
      content: 'Item a4'
    }]
  }, {
    items: [{
      content: 'Item b1'
    }, {
      content: 'Item b2'
    }, {
      content: 'Item b3'
    }, {
      content: 'Item b4'
    }]
  }, {
    items: [{
      content: 'Item c1'
    }, {
      content: 'Item c2'
    }, {
      content: 'Item c3'
    }, {
      content: 'Item c4'
    }]
  }];
