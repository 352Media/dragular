  scope: $scope
});
$scope.$on('drag', function(e, el) {
  e.stopPropagation();
  el.className = el.className.replace(' ex-moved', '');
});
$scope.$on('drop', function(e, el) {
  e.stopPropagation();
  $timeout(function() {
    el.className += ' ex-moved';
  }, 0);
});

// highlight container
$scope.$on('shadow', function highlightContainer(e, item, container) {
  e.stopPropagation();
  $timeout(function() {
    container.className += ' highlighted';
  }, 0);
});
$scope.$on('drag', function highlightRemoval(e, item, container) {
  e.stopPropagation();
  elcontainer.className = container.className.replace(' highlighted', '');
});

  }
});
